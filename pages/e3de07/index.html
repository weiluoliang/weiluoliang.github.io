# n8n安装命令

不使用https请加参数： N8N_SECURE_COOKIE=false

docker volume create n8n_data
docker run -d --name n8n -p 5678:5678 -v n8n_data:/home/node/.n8n -e N8N_SECURE_COOKIE=false  n8nio/n8n
