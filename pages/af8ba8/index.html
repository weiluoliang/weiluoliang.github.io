# 防火墙命令

# 添加
firewall-cmd --zone=public --add-port=6379/tcp --permanent    （--permanent永久生效，没有此参数重启后失效）

# 重新载入
firewall-cmd --reload

# 查看端口是否开放
firewall-cmd --zone= public --query-port=80/tcp

# 删除已开放的端口
firewall-cmd --zone= public --remove-port=80/tcp --permanent
